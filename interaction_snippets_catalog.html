// Lenis + GSAP ScrollTrigger 통합
import Lenis from 'lenis';
import { gsap } from 'gsap';
import { ScrollTrigger } from 'gsap/ScrollTrigger';

gsap.registerPlugin(ScrollTrigger);

const lenis = new Lenis({
  duration: 1.2,
  easing: (t) => Math.min(1, 1.001 - Math.pow(2, -10 * t)),
  smoothWheel: true,
});

// GSAP ticker와 동기화 (중요)
lenis.on('scroll', ScrollTrigger.update);
gsap.ticker.add((time) => {
  lenis.raf(time * 1000);
});
gsap.ticker.lagSmoothing(0);

// 접근성 — prefers-reduced-motion 대응
if (window.matchMedia('(prefers-reduced-motion: reduce)').matches) {
  lenis.destroy();
}

Hover Text Scramble

호버 시 글자가 무작위로 섞이다가 정착하는 효과. 메뉴·CTA에 디지털 디테일을 더한다.

— Mid ★★★
— Hover Me
HOVER ME
// 글자 무작위 → 정착 효과
class TextScramble {
  constructor(el) {
    this.el = el;
    this.chars = '!<>-_\\/[]{}—=+*^?#________';
    this.update = this.update.bind(this);
  }

  setText(newText) {
    const oldText = this.el.innerText;
    const length = Math.max(oldText.length, newText.length);
    const promise = new Promise((resolve) => (this.resolve = resolve));
    this.queue = [];

    for (let i = 0; i < length; i++) {
      const from = oldText[i] || '';
      const to = newText[i] || '';
      const start = Math.floor(Math.random() * 40);
      const end = start + Math.floor(Math.random() * 40);
      this.queue.push({ from, to, start, end });
    }

    cancelAnimationFrame(this.frameRequest);
    this.frame = 0;
    this.update();
    return promise;
  }

  update() {
    let output = '';
    let complete = 0;
    for (let i = 0, n = this.queue.length; i < n; i++) {
      let { from, to, start, end, char } = this.queue[i];
      if (this.frame >= end) {
        complete++;
        output += to;
      } else if (this.frame >= start) {
        if (!char || Math.random() < 0.28) {
          char = this.chars[Math.floor(Math.random() * this.chars.length)];
          this.queue[i].char = char;
        }
        output += `<span style="color:#c9a164">${char}</span>`;
      } else {
        output += from;
      }
    }
    this.el.innerHTML = output;
    if (complete === this.queue.length) this.resolve();
    else { this.frameRequest = requestAnimationFrame(this.update); this.frame++; }
  }
}

// 사용
const el = document.querySelector('[data-scramble]');
const fx = new TextScramble(el);
const original = el.innerText;
el.addEventListener('mouseenter', () => fx.setText(original));

// 마우스 위치 → 버튼이 따라옴
document.querySelectorAll('[data-magnetic]').forEach((btn) => {
  const inner = btn.querySelector('.inner') || btn;
  const strength = parseFloat(btn.dataset.magnetic) || 0.4;

  btn.addEventListener('mousemove', (e) => {
    const rect = btn.getBoundingClientRect();
    const x = e.clientX - rect.left - rect.width / 2;
    const y = e.clientY - rect.top - rect.height / 2;

    gsap.to(btn, {
      x: x * strength,
      y: y * strength,
      duration: 0.6,
      ease: 'power3.out',
    });

    // 내부 텍스트는 더 약하게 (parallax 느낌)
    gsap.to(inner, {
      x: x * strength * 0.6,
      y: y * strength * 0.6,
      duration: 0.6,
      ease: 'power3.out',
    });
  });

  btn.addEventListener('mouseleave', () => {
    gsap.to([btn, inner], {
      x: 0, y: 0,
      duration: 0.8,
      ease: 'elastic.out(1, 0.4)',
    });
  });
});

// Fragment shader — 마우스 위치 기반 ripple
uniform sampler2D uTexture;
uniform vec2 uMouse;
uniform float uHover;
uniform float uTime;
varying vec2 vUv;

void main() {
  vec2 uv = vUv;
  float dist = distance(uv, uMouse);
  float ripple = sin(dist * 30.0 - uTime * 2.0) * 0.02;

  // 마우스에서 멀어질수록 약해짐
  ripple *= smoothstep(0.5, 0.0, dist) * uHover;

  // UV 좌표 왜곡
  uv += ripple * normalize(uv - uMouse);

  gl_FragColor = texture2D(uTexture, uv);
}

Canvas Particle System

마우스에 반응하는 점·선 또는 물리 효과. WebGL 없이 2D Canvas만으로도 충분히 효과적.

— Hard ★★★
— Live 
// Canvas 파티클 + 마우스 인터랙션
const canvas = document.getElementById('particleCanvas');
const ctx = canvas.getContext('2d');
const W = canvas.width;
const H = canvas.height;

const particles = Array.from({ length: 80 }, () => ({
  x: Math.random() * W,
  y: Math.random() * H,
  vx: (Math.random() - 0.5) * 0.4,
  vy: (Math.random() - 0.5) * 0.4,
  r: Math.random() * 1.5 + 0.5,
}));

const mouse = { x: -100, y: -100 };
canvas.addEventListener('mousemove', (e) => {
  const rect = canvas.getBoundingClientRect();
  mouse.x = e.clientX - rect.left;
  mouse.y = e.clientY - rect.top;
});

function render() {
  ctx.clearRect(0, 0, W, H);

  particles.forEach((p, i) => {
    // 마우스에 끌림
    const dx = mouse.x - p.x;
    const dy = mouse.y - p.y;
    const dist = Math.sqrt(dx * dx + dy * dy);

    if (dist < 80) {
      p.vx += (dx / dist) * 0.05;
      p.vy += (dy / dist) * 0.05;
    }

    p.x += p.vx;
    p.y += p.vy;
    p.vx *= 0.98;  // 마찰
    p.vy *= 0.98;

    // 경계 처리
    if (p.x < 0 || p.x > W) p.vx *= -1;
    if (p.y < 0 || p.y > H) p.vy *= -1;

    // 파티클 그리기
    ctx.beginPath();
    ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
    ctx.fillStyle = '#c9a164';
    ctx.fill();

    // 가까운 파티클끼리 선으로 연결
    particles.slice(i + 1).forEach((p2) => {
      const d = Math.hypot(p.x - p2.x, p.y - p2.y);
      if (d < 60) {
        ctx.beginPath();
        ctx.moveTo(p.x, p.y);
        ctx.lineTo(p2.x, p2.y);
        ctx.strokeStyle = `rgba(201, 161, 100, ${1 - d / 60})`;
        ctx.lineWidth = 0.5;
        ctx.stroke();
      }
    });
  });

  requestAnimationFrame(render);
}
render();
